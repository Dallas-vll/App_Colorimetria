import React, { useState, useEffect } from "react";
import { UserAnalysis } from "@/entities/UserAnalysis";
import { Button } from "@/components/ui/button";
import { Link } from "react-router-dom";
import { createPageUrl } from "@/utils";
import { ArrowLeft, Download } from "lucide-react";
import { Skeleton } from "@/components/ui/skeleton";

import SeasonCard from "../components/results/SeasonCard";
import CharacteristicsRadar from "../components/results/CharacteristicsRadar";
import ColorPalettes from "../components/results/ColorPalettes";
import RecommendationsSection from "../components/results/RecommendationsSection";

export default function Results() {
  const [analysis, setAnalysis] = useState(null);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    loadAnalysis();
  }, []);

  const loadAnalysis = async () => {
    const urlParams = new URLSearchParams(window.location.search);
    const id = urlParams.get('id');
    
    if (id) {
      const data = await UserAnalysis.list();
      const found = data.find(a => a.id === id);
      setAnalysis(found);
    }
    setIsLoading(false);
  };

  if (isLoading) {
    return (
      <div className="min-h-screen p-6 md:p-8">
        <div className="max-w-6xl mx-auto space-y-6">
          <Skeleton className="h-12 w-64" />
          <Skeleton className="h-64 w-full" />
          <Skeleton className="h-96 w-full" />
        </div>
      </div>
    );
  }

  if (!analysis) {
    return (
      <div className="min-h-screen p-6 md:p-8 flex items-center justify-center">
        <div className="text-center">
          <h2 className="text-2xl font-bold text-gray-900 mb-4">Análise não encontrada</h2>
          <Link to={createPageUrl("Analysis")}>
            <Button>Fazer Nova Análise</Button>
          </Link>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen p-6 md:p-8">
      <div className="max-w-6xl mx-auto">
        <div className="flex flex-col md:flex-row justify-between items-start md:items-center gap-4 mb-8">
          <div className="flex items-center gap-4">
            <Link to={createPageUrl("MyAnalyses")}>
              <Button variant="outline" size="icon">
                <ArrowLeft className="w-4 h-4" />
              </Button>
            </Link>
            <div>
              <h1 className="text-3xl font-bold text-gray-900">Seus Resultados</h1>
              <p className="text-gray-600 mt-1">Análise personalizada de colorimetria</p>
            </div>
          </div>
          <Button variant="outline" className="gap-2">
            <Download className="w-4 h-4" />
            Exportar Relatório
          </Button>
        </div>

        <div className="grid lg:grid-cols-2 gap-6 mb-8">
          <SeasonCard analysis={analysis} />
          <CharacteristicsRadar analysis={analysis} />
        </div>

        <ColorPalettes analysis={analysis} />
        
        <RecommendationsSection analysis={analysis} />
      </div>
    </div>
  );
}
